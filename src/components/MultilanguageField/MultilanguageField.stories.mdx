# MultilanguageField

Multilanguage field for forms, indicates which fields are multilingual

## Preview of component

<Preview>
	<Story name="Base">
		<Formik onSubmit={console.log} initialValues={{label: { nl: 'nederlands', multilanguage: true }}}>
			{({ errors, submitForm, values, resetForm, setFieldValue }) => {
				return (
					<MultilanguageField
						as={TextField}
						label="Omschrijving"
						name="description"
						value={values.description}
						setFieldValue={setFieldValue}
						languages={['nl']}
						activeLanguage={'nl'}/>
				)
			}}
		</Formik>
	</Story>
